const pool = new Pool({
  host:     process.env.DB_HOST,      // 'db' aus .env

src/index.js
javascript

const http  = require('http');
const redis = require('redis');
const pool  = require('./db');

// REDIS_URL kommt aus .env: redis://cache:6379
const redisClient = redis.createClient({ url: process.env.REDIS_URL });
redisClient.on('error', err => console.error('Redis error:', err));
redisClient.connect().catch(console.error);

const server = http.createServer(async (req, res) => {
  if (req.url === '/health') {
    res.writeHead(200); res.end('OK');
    return;
  }
  try {
    const cached = await redisClient.get('visits');
    if (cached) {
      res.writeHead(200); res.end(`Visits (cached): ${cached}`);
      return;
    }
    const { rows } = await pool.query('SELECT COUNT(*) FROM visits');
    const count = rows[0].count;
    await redisClient.setEx('visits', 5, count);
    res.writeHead(200); res.end(`Visits (db): ${count}`);
  } catch (err) {
    console.error(err);
    res.writeHead(500); res.end('Internal Server Error');
  }
});

server.listen(3000, () => console.log('Server ready on :3000'));

1.6 docker-compose.yml (korrigiert)
yaml

# Kein 'version:'-Schlüssel nötig — Compose v2 Plugin erkennt Format automatisch
services:
  app:
    build: .
    ports: ["3000:3000"]
    env_file: .env
    depends_on:
      db:
        condition: service_healthy
      cache:
        condition: service_started
    networks: [backend]
    restart: on-failure

  db:
    image: postgres:16-alpine
    environment:
      POSTGRES_USER:     ${DB_USER}
      POSTGRES_PASSWORD: ${DB_PASSWORD}
      POSTGRES_DB:       ${DB_NAME}
    volumes:
      - pgdata:/var/lib/postgresql/data
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql:ro
    healthcheck:
      # Doppeltes $$ verhindert Shell-Expansion durch Compose –
      # der Wert wird wörtlich an den Container übergeben
      test: ["CMD-SHELL", "pg_isready -U $$POSTGRES_USER -d $$POSTGRES_DB"]
      interval: 5s
      timeout: 3s
      retries: 5
      start_period: 10s
    networks: [backend]

  cache:
    image: redis:7-alpine
    command: redis-server --save "" --appendonly no
    networks: [backend]

volumes:
  pgdata:

networks:
  backend:
    driver: bridge
    # HINWEIS: internal: true blockiert ausgehende Internetverbindungen.
    # Sinnvoll für Produktionsumgebungen, kann beim ersten Testen
    # externe npm-Registry-Zugriffe verhindern – daher hier auskommentiert.
    # internal: true

Element	Detail Korrektur v2
depends_on condition: service_healthy ist in Compose v2 Plugin gültig, aber nur wenn der referenzierte Service einen funktionierenden Healthcheck hat Healthcheck schlug wegen wget-Fehler immer fehl (siehe 1.5) Healthcheck korrigiert

init.sql wurde nicht eingebunden Tabelle fehlte	Volume-Mount hinzugefügt
internal: true war für Einsteiger verwirrend – Auskommentiert und erklärt

2.2 Pipeline: .github/workflows/container-ci.yml
yaml

name: Container CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

env:
  REGISTRY: ghcr.io
  # github.repository liefert 'org/repo' – muss Kleinbuchstaben sein.
  # Der Schritt 'Normalize image name' wandelt das sicher um.
  IMAGE_NAME: ${{ github.repository }}

jobs:
  build-scan-push:
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write
      security-events: write

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      # KORRIGIERT: Repository-Namen in Kleinbuchstaben umwandeln.
      # GitHub erlaubt Großbuchstaben in Repo-Namen, GHCR verlangt lowercase.
      - name: Normalize image name
        id: image
        run: |
          NAME=$(echo "${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}" | tr '[:upper:]' '[:lower:]')
          echo "name=$NAME" >> $GITHUB_OUTPUT

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Build image (local, no push)
        uses: docker/build-push-action@v5
        with:
          context: .
          load: true
          tags: ${{ steps.image.outputs.name }}:${{ github.sha }}
          cache-from: type=gha
          cache-to:   type=gha,mode=max

      - name: Trivy vulnerability scan
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: ${{ steps.image.outputs.name }}:${{ github.sha }}
          format: sarif
          output: trivy-results.sarif
          severity: CRITICAL,HIGH
          exit-code: 1

      - name: Upload Trivy results to GitHub Security
        uses: github/codeql-action/upload-sarif@v3
        if: always()
        with:
          sarif_file: trivy-results.sarif

      - name: Login to GHCR
        if: github.ref == 'refs/heads/main'
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Push image
        if: github.ref == 'refs/heads/main'
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: |
            ${{ steps.image.outputs.name }}:latest
            ${{ steps.image.outputs.name }}:${{ github.sha }}
          cache-from: type=gha

Element	Detail	Korrektur v2
IMAGE_NAME enthielt github.repository direkt – Großbuchstaben möglich GHCR lehnt Image-Namen mit Großbuchstaben ab (HTTP 400) Neuer Schritt Normalize image name mit tr '[:upper:]' '[:lower:]'
Alle nachfolgenden Schritte nutzen steps.image.outputs.name statt env.IMAGE_NAME – Einheitlich umgestellt
⚠️ Fehlerbild: Pipeline bricht durch Basis-Image-CVEs ab

    Symptom: trivy-action gibt exit-code 1, obwohl der eigene Code sauber ist.

    Ursache: Ungepatchte CVEs im Basis-Image (z. B. node:20-alpine).

    Lösungen (nach Priorität):

        Neuestes Patch-Image pinnen: node:20.14-alpine3.20

        .trivyignore-Datei für akzeptierte CVEs mit Ablaufdatum anlegen:
        text

        # .trivyignore
        CVE-2024-XXXXX  # Expires: 2026-12-31, kein Fix verfügbar

        Distroless als Laufzeit-Base: gcr.io/distroless/nodejs20-debian12 (drastisch kleinere Angriffsfläche, kein Shell, kein Paketmanager)

3.1 docker inspect – strukturierte Metadaten

# Vollständige JSON-Ausgabe

# Exit-Code des letzten Container-Starts

# 0=OK  1=App-Fehler  137=OOM-Kill  143=SIGTERM

Exit-Code: Bedeutung
0 Normales Ende
1 Applikationsfehler
126 Befehl nicht ausführbar
127 Befehl nicht gefunden
137 OOM-Kill (SIGKILL)
143 Graceful Shutdown (SIGTERM)
3.2 nsenter – Namespaces betreten

nsenter ermöglicht es, in Namespaces eines laufenden Containers einzutreten, ohne dass das Container-Image eine Shell enthalten muss. Es wird die Shell des Hosts verwendet.
bash

PID=$(docker inspect -f '{{.State.Pid}}' <container-id>)

# Alle Namespaces betreten (PID, Net, Mount, UTS, IPC)
sudo nsenter -t $PID --all -- /bin/sh

# Nur Netzwerk-Namespace: ip addr, ss -tulpn
sudo nsenter -t $PID --net -- ip addr show

# Dateisystem inspizieren (funktioniert auch ohne Shell im Image)
sudo nsenter -t $PID --mount -- ls /app

# Prozessliste im PID-Namespace
sudo nsenter -t $PID --pid -- ps aux

Wann nsenter statt docker exec?

    Problem: docker exec startet einen neuen Prozess im Container – schlägt fehl, wenn der Container crasht, in einer Crash-Loop ist oder kein /bin/sh enthält.

    Lösung: nsenter agiert auf Kernel-Ebene und ist unabhängig vom Container-Status.

    Anwendungsfälle: Distroless Images, hängende Prozesse, Crash-Loop-Debugging.

3.3 cgroups v2 – Ressourcenlimits direkt lesen
bash

PID=$(docker inspect -f '{{.State.Pid}}' <container-id>)

# cgroup-Pfad ermitteln (unter cgroups v2 beginnt die Zeile mit '0::')
CGROUP=$(cat /proc/$PID/cgroup | head -1 | cut -d: -f3)
BASE=/sys/fs/cgroup${CGROUP}

cat $BASE/memory.max      # RAM-Limit ('max' = unbegrenzt)
cat $BASE/memory.current  # Aktuelle Nutzung in Bytes
cat $BASE/memory.events | grep oom_kill  # OOM-Zähler
cat $BASE/cpu.max         # CPU-Quota: '50000 100000' = 50% einer CPU
cat $BASE/io.stat         # I/O-Statistiken
cat $BASE/cgroup.procs    # Alle PIDs im cgroup

💡 Praxis-Tipp: OOM-Debugging (Exit-Code 137)

    docker inspect -f '{{.State.OOMKilled}}' <id> → true/false

    cat $BASE/memory.events | grep oom_kill → Zähler > 0

    dmesg | grep -i 'oom\|killed' → Kernel-Log

    Lösung A: --memory-Limit erhöhen (docker run --memory=512m ...)

    Lösung B: Speicherleck im Code beheben (Heap-Profiling mit node --inspect)

Aufgabe 1 – LF 10b: Docker Compose Fehleranalyse

Aufgabenstellung
yaml

services:
  web:
    image: nginx:latest
    ports: ["80:80"]
    depends_on: [db]
  db:
    image: postgres:16
    environment:
      POSTGRES_PASSWORD: geheim123
    volumes:
      - ./data:/var/lib/postgresql/data:ro
    network: backend
networks:
  backend:

Musterlösung – 3 Fehler

    network: backend ist falsch (kein gültiger Compose-Schlüssel).
    Korrektur: networks: [backend]

    :ro (read-only) auf dem PostgreSQL-Datenverzeichnis. PostgreSQL muss schreiben können – der Container startet nie korrekt.
    Korrektur: - ./data:/var/lib/postgresql/data

    depends_on: [db] ohne Healthcheck garantiert nur, dass db gestartet wurde, nicht dass PostgreSQL bereit ist.
    Korrektur: depends_on: db: condition: service_healthy + healthcheck am db-Service definieren (pg_isready).

Aufgabe 2 – LF 11b: Sicherheitskonzept bewerten

Aufgabenstellung
bash

docker run -d --privileged \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -e API_KEY=sk-prod-abc123xyz \
  --network host \
  myapp:latest

Musterlösung – 4 kritische Verstösse

    --privileged: Vollzugriff auf alle Kernel-Capabilities des Hosts. Entspricht root-Zugriff auf den Host-Kernel.
    Fix: --cap-drop=ALL --cap-add=NET_BIND_SERVICE

    docker.sock-Mount: Wer diesen Socket hat, kontrolliert den Daemon und damit den gesamten Host – privilege escalation in einem Schritt.
    Fix: Socket niemals in Anwendungscontainer mounten.

    API_KEY als ENV: Sichtbar via docker inspect für alle Host-User.
    Fix: Docker Secrets (Swarm) oder externe Vault-Integration.

    --network host: Hebt Netzwerk-Isolation vollständig auf.
    Fix: Benutzerdefiniertes Bridge-Netzwerk vorab anlegen:
    docker network create app-net → docker run ... --network app-net ...
